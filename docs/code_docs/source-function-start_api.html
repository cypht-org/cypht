 14: class Hm_Handler_swipe_2fa_check extends Hm_Handler_Module {
 15:     public function process() {
 16: 
 17:         /* new session or one not passed the second auth */
 18:         if ($this->session->loaded || $this->session->get('2fa_required', false)) {
 19: 
 20:             /* ini file location */
 21:             $ini_file = rtrim($this->config->get('app_data_dir', ''), '/').'/swipeidentity.ini';
 22: 
 23:             /* data for the swipe api */
 24:             $swipe_username = $this->session->get('username', false);
 25:             $swipe_address = $this->request->server['REMOTE_ADDR'];
 26:             $required = true;
 27: 
 28:             /* get api config and object */
 29:             list($api, $api_config) = setup_swipe_api($ini_file);
 30:             $started = start_api($api, $api_config);
 31:             if (!$started) {
 32:                 $this->out('2fa_fatal', true);
 33:             }
 34: 
 35:             /* get current 2fa state */
 36:             if (!array_key_exists('2fa_sms_response', $this->request->post)) {
 37:                 $state = get_secondfactor_state($api, $api_config, $swipe_username, $swipe_address);
 38:             }
 39:             else {
 40:                 $state = RC_SMS_DELIVERED;
 41:             }
 42: 
 43:             /* pass a key and no redirect flag to the output modules */
 44:             $this->out('no_redirect', true);
 45:             Hm_Request_Key::load($this->session, $this->request, false);
 46:             $this->out('2fa_key', Hm_Request_Key::generate());
 47: 
 48:             $sms_number = false;
 49:             $sms_response = false;
 50: 
 51:             /* if the user has not registered a phone number yet look for one in POST */
 52:             if ($state == NEED_REGISTER_SMS && array_key_exists('sms_number', $this->request->post)) {
 53: 
 54:                 /* remove non numeric delimiters */
 55:                 $sms_number = preg_replace("/[^\d]/", "", $this->request->post['sms_number']);
 56: 
 57:                 /* US phone numbers only for now */
 58:                 if (preg_match("/^1\d{10}$/", $sms_number)) {
 59:                     $submit_number = $sms_number;
 60: 
 61:                     /* set the phone number using the api */
 62:                     $api->setUserSmsNumber($swipe_username, $api_config["com.swipeidentity.api.appcode"], $submit_number);
 63: 
 64:                     /* refecth the status */
 65:                     $state = get_secondfactor_state($api, $api_config, $swipe_username, $swipe_address);
 66: 
 67:                     /* number rejected by swipe */
 68:                     if ($state == NEED_REGISTER_SMS) {
 69:                         $this->out('2fa_error', 'Invalid phone number');
 70:                     }
 71:                 }
 72:                 else {
 73:                     $this->out('2fa_error', 'Invalid phone number format');
 74:                 }
 75: 
 76:             }
 77:             /* the sms was sent, look for a sms code in POST */
 78:             elseif ($state == RC_SMS_DELIVERED && array_key_exists('2fa_sms_response', $this->request->post)) {
 79: 
 80:                 if (preg_match("/^\d{5}$/", $this->request->post['2fa_sms_response'])) {
 81: 
 82:                     $sms_response = $this->request->post['2fa_sms_response'];
 83: 
 84:                     /* validate the sms response with the api */
 85:                     $resp = $api->answerSMS($swipe_username, $api_config["com.swipeidentity.api.appcode"], $sms_response);
 86: 
 87:                     /* success! allow the user to login */
 88:                     if ($resp->getReturnCode() == RC_SMS_ANSWER_ACCEPTED) {
 89:                         $required = false;
 90:                     }
 91:                     else {
 92:                         $state = get_secondfactor_state($api, $api_config, $swipe_username, $swipe_address);
 93:                         $this->out('2fa_error', 'Response did not match! A new sms code has been sent');
 94:                     }
 95:                 }
 96:                 else {
 97:                     $this->out('2fa_error', 'Incorrectly formatted response, please re-enter the sms code');
 98:                 }
 99:             }
100: 
101:             /* if required is true we still have not completed the 2fa */
102:             if ($required) {
103: 
104:                 /* pass required flag to modules */
105:                 $this->session->set('2fa_required', true);
106:                 $this->out('2fa_required', true);
107:                 $this->out('2fa_state', $state);
108: 
109:                 /* close the session early */
110:                 $this->session->close_early();
111:             }
112:             else {
113: 
114:                 /* unset any previously set required flags */
115:                 $this->session->set('2fa_required', false);
116:                 $this->out('2fa_required', false);
117:             }
118:         }
119:     }
120: }
