 95: /**
 96:  * Build a human readable interval string
 97:  * @subpackage core/functions
 98:  * @param string $date_str date string parsable by strtotime()
 99:  * @return string
100:  */
101: function human_readable_interval($date_str) {
102:     $precision     = 2;
103:     $interval_time = array();
104:     $date          = strtotime($date_str);
105:     $interval      = time() - $date;
106:     $res           = array();
107: 
108:     $t['second'] = 1;
109:     $t['minute'] = $t['second']*60;
110:     $t['hour']   = $t['minute']*60;
111:     $t['day']    = $t['hour']*24;
112:     $t['week']   = $t['day']*7;
113:     $t['month']  = $t['day']*30;
114:     $t['year']   = $t['week']*52;
115: 
116:     if ($interval < 0) {
117:         return 'From the future!';
118:     }
119:     elseif ($interval == 0) {
120:         return 'Just now';
121:     }
122:     foreach (array_reverse($t) as $name => $val) {
123:         if ($interval_time[$name] = ($interval/$val > 0) ? floor($interval/$val) : false) {
124:             $interval -= $val*$interval_time[$name];
125:         }
126:     }
127:     $interval_time = array_slice(array_filter($interval_time, function($v) { return $v > 0; }), 0, $precision);
128:     foreach($interval_time as $name => $val) {
129:         if ($val > 1) {
130:             $res[] = sprintf('%d %ss', $val, $name);
131:         }
132:         else {
133:             $res[] = sprintf('%d %s', $val, $name);
134:         }
135:     }
136:     return implode(', ', $res);
137: }

432:     $links = '';

456:     }
