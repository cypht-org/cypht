 24:      * Assign site config
 25:      * @param object $config site config

 27:      */
 28:     public function __construct($config) {

 30:     }
 31: 
 32:     /**
 33:      * This is the method new auth mechs need to override.

 80:     static public $internal_users = true;
 81: 
 82:     /**
 83:      * Send the username and password to the configured DB for authentication
 84:      * @param string $user username
 85:      * @param string $pass password
 86:      * @return bool true if authentication worked
 87:      */
 88:     public function check_credentials($user, $pass) {
 89:         if ($this->connect()) {
 90:             $sql = $this->dbh->prepare("select hash from hm_user where username = ?");
 91:             if ($sql->execute(array($user))) {
 92:                 $row = $sql->fetch();
 93:                 if ($row['hash'] && Hm_Crypt::check_password($pass, $row['hash'])) {
 94:                     return true;
 95:                 }
 96:             }
 97:         }
 98:         sleep(2);
 99:         return false;
100:     }
101: 
102:     /**
103:      * Delete a user account from the db
104:      * @param string $user username
105:      * @return bool true if successful
106:      */
107:     public function delete($user) {
108:         if ($this->connect()) {
109:             $sql = $this->dbh->prepare("delete from hm_user where username = ?");
110:             if ($sql->execute(array($user)) && $sql->rowCount() == 1) {
111:                 return true;
112:             }
113:         }
114:         return false;
115:     }
116: 
117:     /**
118:      * Create a new or re-use an existing DB connection
119:      * @return bool true if the connection is available
120:      */
121:     protected function connect() {
122:         $this->dbh = Hm_DB::connect($this->site_config);
123:         if ($this->dbh) {
124:             return true;
125:         }
126:         return false;
127:     }
128: 
129:     /**
130:      * Change the password for a user in the DB
131:      * @param string $user username
132:      * @param string $pass password
133:      * @return bool true on success
134:      */
135:     public function change_pass($user, $pass) {
136:         $this->connect();
137:         $hash = Hm_Crypt::hash_password($pass);
138:         $sql = $this->dbh->prepare("update hm_user set hash=? where username=?");
139:         if ($sql->execute(array($hash, $user)) && $sql->rowCount() == 1) {
140:             Hm_Msgs::add("Password changed");
141:             return true;
142:         }
143:         return false;
144:     }

154:         $this->connect();
