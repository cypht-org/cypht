 10: /**
 11:  * Struct that makes it easy for a module set to manage a list of server connections
 12:  */
 13: trait Hm_Server_List {
 14: 
 15:     /* list of server connections */
 16:     private static $server_list = array();
 17: 
 18:     /**
 19:      * Server lists must override this method to connect
 20:      * @param int $id server id
 21:      * @param string $server server hostname or ip
 22:      * @param string $user username for authentication
 23:      * @param string $pass password for authentication
 24:      * @param mixed $cache cached connection data
 25:      * @return bool true on success
 26:      */
 27:     abstract public function service_connect($id, $server, $user, $pass, $cache);
 28: 
 29:     /**
 30:      * Connect to a server
 31:      * @param int $id server id
 32:      * @param mixed $cache cached server data
 33:      * @param string $user username
 34:      * @param string $pass password
 35:      * @param bool $save_credentials true to save the username and password
 36:      * @return mixed connection object on success, otherwise false
 37:      */
 38:     public static function connect($id, $cache=false, $user=false, $pass=false, $save_credentials=false) {
 39:         if (array_key_exists($id, self::$server_list)) {
 40:             $server = self::$server_list[$id];
 41:             if ($server['object']) {
 42:                 return $server['object'];
 43:             }
 44:             else {
 45:                 if ((!$user || !$pass) && (!array_key_exists('user', $server) || !array_key_exists('pass', $server))) {
 46:                     return false;
 47:                 }
 48:                 elseif (array_key_exists('user', $server) && array_key_exists('pass', $server)) {
 49:                     $user = $server['user'];
 50:                     $pass = $server['pass'];
 51:                 }
 52:                 $res = self::service_connect($id, $server, $user, $pass, $cache);
 53:                 if ($res) {
 54:                     self::$server_list[$id]['connected'] = true;
 55:                     if ($save_credentials) {
 56:                         self::$server_list[$id]['user'] = $user;
 57:                         self::$server_list[$id]['pass'] = $pass;
 58:                     }
 59:                 }
 60:                 return self::$server_list[$id]['object'];
 61:             }
 62:         }
 63:         return false;
 64:     }
 65: 
 66:     /**
 67:      * Remove the username and password from a connection
 68:      * @param int $id server id
 69:      * @return void
 70:      */
 71:     public static function forget_credentials($id) {
 72:         if (array_key_exists($id, self::$server_list)) {
 73:             unset(self::$server_list[$id]['user']);
 74:             unset(self::$server_list[$id]['pass']);
 75:         }
 76:     }
 77: 
 78:     /**
 79:      * Add a server definition
 80:      * @param array $atts server details
 81:      * @param int $id server id
 82:      * @return void
 83:      */
 84:     public static function add($atts, $id=false) {
 85:         $atts['object'] = false;
 86:         $atts['connected'] = false;
 87:         if ($id !== false) {
 88:             self::$server_list[$id] = $atts;
 89:         }
 90:         else {
 91:             self::$server_list[] = $atts;
 92:         }
 93:     }
 94: 
 95:     /**
 96:      * Remove a server
 97:      * @param int $id server id
 98:      * @return bool true on success
 99:      */
100:     public static function del($id) {
101:         if (array_key_exists($id, self::$server_list)) {
102:             unset(self::$server_list[$id]);
103:             return true;
104:         }
105:         return false;
106:     }
107: 
108:     /**
109:      * Return all server details
110:      * @param int $id if not false, return details for this server only
111:      * @param bool $full true to return passwords for server connections. CAREFUL!
112:      * @return array server details
113:      */
114:     public static function dump($id=false, $full=false) {
115:         $list = array();
116:         foreach (self::$server_list as $index => $server) {
117:             if ($id !== false && $index != $id) {
118:                 continue;
119:             }
120:             if ($full) {
121:                 $list[$index] = $server;
122:             }
123:             else {
124:                 $list[$index] = array(
125:                     'name' => $server['name'],
126:                     'server' => $server['server'],
127:                     'port' => $server['port'],
128:                     'tls' => $server['tls']
129:                 );
130:                 if (array_key_exists('user', $server)) {
131:                     $list[$index]['user'] = $server['user'];
132:                 }
133:             }
134:             if ($id !== false) {
135:                 return $list[$index];
136:             }
137:         }
138:         return $list;
139:     }
140: 
141:     /**
142:      * Try to disconnect cleanly
143:      * @param int $id server id
144:      * @return void
145:      */
146:     public static function clean_up($id=false) {
147:         if ($id !== false && array_key_exists($id, self::$server_list)) {
148:             self::disconnect($id);
149:         }
150:         else {
151:             foreach (self::$server_list as $index => $server) {
152:                 self::disconnect($index);
153:             }
154:         }
155:     }
156: 
157:     /**
158:      * Disconnect from a server
159:      * @param int $id the server id to disconnect
160:      * @return void
161:      */
162:     public static function disconnect($id) {
163:         if (self::$server_list[$id]['connected'] && self::$server_list[$id]['object']) {
164:             if (method_exists(self::$server_list[$id]['object'], 'disconnect')) {
165:                 self::$server_list[$id]['object']->disconnect();
166:             }
167:             self::$server_list[$id]['connected'] = false;
168:         }
169:     }
170: }
