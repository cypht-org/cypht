 383:                     $this->user_config->save($user, $pass);
 384:                     $this->session->set('changed_settings', array());
 385:                     if ($logout) {
 386:                         $this->session->destroy($this->request);
 387:                         Hm_Msgs::add('Saved user data on logout');
