 58: 
 59:     /* POP3 authentication server settings */
 60:     private $pop3_settings = array();
 61: 
 62:     /**
 63:      * Send the username and password to the configured POP3 server for authentication
 64:      * @param $user string username
 65:      * @param $pass string password
 66:      * @return bool true if authentication worked
 67:      */
 68:     public function check_credentials($user, $pass) {
 69:         $pop3 = new Hm_POP3();
 70:         $authed = false;
 71:         list($server, $port, $tls) = $this->get_pop3_config();
 72:         if ($user && $pass && $server && $port) {
 73:             $this->pop3_settings = array(
 74:                 'server' => $server,
 75:                 'port' => $port,
 76:                 'tls' => $tls,
 77:                 'username' => $user,
 78:                 'password' => $pass,
 79:                 'no_caps' => true
 80:             );
 81:             $pop3->server = $server;
 82:             $pop3->port = $port;
 83:             $pop3->tls = $tls;
 84:             if ($pop3->connect()) {
 85:                 $authed = $pop3->auth($user, $pass);
 86:             }
 87:         }
 88:         if ($authed) {
 89:             return true;
 90:         }
 91:         Hm_Msgs::add("Invalid username or password");

139:         $this->debug = array();
140:         $this->server = 'localhost';
141:         $this->port = 110; // ssl @ 995
142:         $this->ssl = false;
143:         $this->starttls = false;

153:     /**
154:      * Get server response
155:      * @param bool $multi_line multi-line flag
156:      * @return string
157:      */
158:     private function get_response($multi_line=false) {
159:         if ($multi_line) {
160:             $res = $this->get_multi_line_response();
161:         }
162:         else {
163:             $res = array($this->get_single_line_response());

168:     /**
169:      * Read in a multi-line response
170:      * @param int $line_length max read length
171:      * @return string
172:      */
173:     private function get_multi_line_response($line_length=8192) {
174:         $n = -1;
175:         $result = array();
176:         if (!is_resource($this->handle)) {
177:             return $result;
178:         }
179:         do {
180:             if ($n > 0 && $result[$n] == "..\r\n") {
181:                 $result[$n] = ".\r\n";
182:             }
183:             $n++;
184:             $result[$n] = fgets($this->handle, $line_length);
185:             while(substr($result[$n], -2) != "\r\n") {
186:                 if (!is_resource($this->handle)) {
187:                     break;
188:                 }
189:                 $result[$n] .= fgets($this->handle, $line_length);
190:             }
191:             if ($this->is_error($result)) {
