 16:      * update the cache untagged QRESYNC FETCH responses
 17:      * @param array $data low level parsed IMAP response segment
 18:      * @return int 1 if the cache was updated
 19:      */
 20:     protected function update_cache_data($data) {
 21:         $res = 0;
 22:         if (in_array('VANISHED', $data)) {
 23:             $uid = $this->get_adjacent_response_value($data, -1, 'VANISHED');
 24:             if ($this->is_clean($uid, 'uid')) {
 25:                 if (isset($this->cache_data[$this->selected_mailbox['name']])) {
 26:                     $key = $this->selected_mailbox['name'];
 27:                     foreach ($this->cache_data[$key] as $command => $result) {
 28:                         if (strstr($command, 'UID FETCH')) {
 29:                             if (isset($result[$uid])) {
 30:                                 unset($this->cache_data[$key][$command][$uid]);
 31:                                 $this->debug[] = sprintf('Removed message from cache using QRESYNC response (uid: %s)', $uid);
 32:                                 $res = 1;
 33:                             }
 34:                         }
 35:                         elseif (strstr($command, 'UID SORT')) {
 36:                             $index = array_search($uid, $result);
