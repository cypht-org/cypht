  9: /**
 10:  * Base class that controls how data is output
 11:  * @abstract
 12:  */
 13: abstract class Hm_Output {
 14: 
 15:     /**
 16:      * Extended classes must override this method to output content
 17:      * @param mixed $content data to output
 18:      * @return void
 19:      */
 20:     abstract protected function output_content($content);
 21: 
 22:     /**
 23:      * Wrapper around extended class output_content() calls
 24:      * @param mixed $response data to output
 25:      * @param array $input raw module data
 26:      * @return void
 27:      */
 28:     public function send_response($response, $input=array()) {
 29:         if (array_key_exists('http_headers', $input)) {
 30:             $this->output_content($response, $input['http_headers']);
 31:         }
 32:         else {
 33:             $this->output_content($response);
 34:         }
 35:     }
 36: }
 37: 
 38: /**
 39:  * Output request responses using HTTP
 40:  */
 41: class Hm_Output_HTTP extends Hm_Output {
 42: 
 43:     /**
 44:      * Send HTTP headers
 45:      * @param array $headers headers to send
 46:      * @return void
 47:      */
 48:     protected function output_headers($headers) {
 49:         foreach ($headers as $header) {
 50:             Hm_Functions::header($header);
 51:         }
 52:     }
 53: 
 54:     /**
 55:      * Send response content to the browser
 56:      * @param mixed $content data to send
 57:      * @param array $headers HTTP headers to set
 58:      * @return void
 59:      */
 60:     protected function output_content($content, $headers=array()) {
 61:         $this->output_headers($headers);
 62:         ob_end_clean();
 63:         echo $content;
 64:     }
 65: }

115: trait Hm_List {

121:      * Add a message
122:      * @param string $string message to add
123:      * @return void
124:      */
125:     public static function add($string) {
126:         self::$msgs[] = self::str($string, false);
127:     }
128: 
129:     /**
130:      * Return all messages
131:      * @return array all messages
132:      */
133:     public static function get() {
134:         return self::$msgs;
135:     }
136: 
137:     /**
138:      * Stringify a value
139:      * @param mixed $mixed value to stringify
140:      * @return string
141:      */
142:     public static function str($mixed, $return_type=true) {
143:         $type = gettype($mixed);
144:         if (in_array($type, array('array', 'object'), true)) {
145:             $str = print_r($mixed, true);
146:         }
147:         elseif ($return_type) {
148:             $str = sprintf("%s: %s", $type, $mixed);
149:         }
150:         else {
151:             $str = (string) $mixed;
152:         }
153:         return $str;
154:     }
155: 
156:     /**
157:      * Show all messages
158:      * @param string $type can be one of "print", "log", or "return"
159:      * @return mixed
160:      */
161:     public static function show($type='print') {
162:         if ($type == 'log') {
163:             Hm_Functions::error_log(print_r(self::$msgs, true));
164:         }
165:         elseif ($type == 'return') {
166:             return self::$msgs;
167:         }
168:         else {
169:             print_r(self::$msgs);
170:         }
171:     }
172: }
