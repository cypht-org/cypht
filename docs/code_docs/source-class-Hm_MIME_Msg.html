 41: /**
 42:  * Connect to and interact with SMTP servers
 43:  * @subpackage smtp/lib
 44:  */
 45: class Hm_SMTP {
 46:     private $config;
 47:     private $server;
 48:     private $starttls;
 49:     private $port;
 50:     private $tls;
 51:     private $auth;
 52:     private $handle;
 53:     private $debug;
 54:     private $hostname;
 55:     private $command_count;
 56:     private $commands;
 57:     private $responses;
 58:     private $smtp_err;
 59:     private $banner;
 60:     private $capability;
 61:     private $connected;
 62:     private $crlf;

 67:     private $request_auths = array();
 68: 
 69:     function __construct($conf) {
 70: 
 71:         $this->hostname = php_uname('n');
 72:         if (preg_match("/:\d+$/", $this->hostname)) {
 73:             $this->hostname = substr($this->hostname, 0, strpos($this->hostname, ':'));
 74:         }
 75:         $this->debug = array();
 76:         if (isset($conf['server'])) {
 77:             $this->server = $conf['server'];
 78:         }
 79:         else {

 81:         }
 82:         if (isset($conf['port'])) {
 83:             $this->port = $conf['port'];
 84:         }
 85:         else {
 86:             $this->port = 25;
 87:         }
 88:         if (isset($conf['tls']) && $conf['tls']) {
 89:             $this->tls = true;
 90:         }
 91:         else {
 92:             $this->tls = false;
 93:         }
 94:         if (!$this->tls) {
 95:             $this->starttls = true;
 96:         }
 97:         $this->request_auths = array('cram-md5', 'login', 'plain');
 98:         if (isset($conf['auth'])) {
 99:             array_unshift($this->request_auths, $conf['auth']);
100:         }
101:         $this->smtp_err = '';
102:         $this->supports_tls = false;
103:         $this->auth = true;
104:         $this->supports_auth = array();
105:         $this->handle = false;
106:         $this->state = 'started';
107:         $this->command_count = 0;
108:         $this->commands = array();
109:         $this->responses = array();
110:         $this->banner = '';
111:         $this->crlf = "\r\n";
112:         $this->capability = '';
113:         $this->line_length = 2048;
114:         $this->connected = false;

117:         $this->max_message_size = 0;

120:     /* send command to the server. Append "\r\n" to the end. */
121:     function send_command($command) {
122:         if (is_resource($this->handle)) {
123:             fputs($this->handle, $command.$this->crlf);
124:         }
125:         $this->commands[] = trim($command);
126:     }

128:     /* loop through "lines" returned from smtp and parse
129:        them. It can return the lines in a raw format, or 
130:        parsed into atoms. 
131:     */
132:     function get_response($chunked=true) {
133:         $n = -1;
134:         $result = array();
135:         $chunked_result = array();
136:         do {
137:             $n++;
138:             if (!is_resource($this->handle)) {
139:                 break;
140:             }
141:             $result[$n] = fgets($this->handle, $this->line_length);
142:             $chunks = $this->parse_line($result[$n]);
143:             if ($chunked) {
144:                 $chunked_result[] = $chunks;
145:             }
146:             if (!trim($result[$n])) {
147:                 unset($result[$n]);
148:                 break;
149:             }
150:             $cont = false;
151:             if (strlen($result[$n]) > 3 && substr($result[$n], 3, 1) == '-') {
152:                 $cont = true;
153:             }
154:         } while ($cont);
155:         $this->responses[] = $result;
156:         if ($chunked) {
157:             $result = $chunked_result;
158:         }
159:         return $result;
160:     }
161: 
162:     /* parse out a line */
163:     function parse_line($line) {
164:         $parts = array();
165: 
166:         $code = substr($line, 0, 3);
167:         $parts[] = $code;
168: 
169:         $remainder = explode(' ',substr($line, 4));
170:         $parts[] = $remainder;
171: 
172:         return $parts;
173:         
174:     }
175:     /* Checks if the numeric response matches the code in $check. 
176:        The return value is simalar to strcmp
177:        Returns <0 if $check is less than the response
178:        Returns  0 if $check is equal to the response
179:        Returns >0 if $check is greater than the response
180:     */
181:     function compare_response($chunked_response, $check) {
182:         $size = count($chunked_response);
183:         if ($size) {

387:     function parse_ntlm_type_two($bin_str) {
388:         $res = array();
389:         $res['vals'] = unpack('a8prefix/Vtype/vname_len/vname_space/Vname_offset/Vflags/A8challenge/A8context/vtarget_len/vtarget_space/Vtarget_offset', base64_decode($bin_str));

487:         $key = array();
488:         $tmp = array();

492:         $key[] = $tmp[0] & 254;
493:         $key[] = ($tmp[0] << 7) | ($tmp[1] >> 1);

497:         $key[] = ($tmp[4] << 3) | ($tmp[5] >> 5);

618:     /* try to accurately validate an E-mail. Based on RFC 3696 */
619:     function match_email_full($val) {
620:         /* defaults */
621:         $domain = false;
622:         $local = false;
623:         /* basic checks to weed out obviously incorrect values */
624:         if (!trim($val) || strlen($val) > 320) {
625:             return false;
626:         }
627:         /* determine if this is a local address or if it has a domain part */
628:         if (strpos($val, '@') !== false) {
629:             $local = substr($val, 0, strrpos($val, '@'));
630:             $domain = substr($val, (strrpos($val, '@') + 1));
631:         }
632:         else {
633:             $local = $val;
634:         }
635:         /* domain is not require but the local part is */
636:         if (!$local) {
637:             return false;
638:         }
639:         else {
640:             /* if we have a domain validate it. */
641:             if ($domain && !$this->validate_domain_full($domain)) {
642:                 return false;
643:             }
644:             /* validate the required local part */
645:             if (!$this->validate_local_full($local)) {
646:                 return false;
647:             }
648:         }
649:         /* E-mail is valid */
650:         return true;
651:     }

653:     function validate_local_full($val) {
654:         /* check length, "." rules, and for characters > ASCII 127 */
655:         if (strlen($val) > 64 || $val{0} == '.' || $val{(strlen($val) -1)} == '.' || strstr($val, '..') ||
656:             preg_match('/[^\x00-\x7F]/',$val)) {
657:             return false;
658:         }
659:         /* remove escaped characters and quoted strings */
660:         $local = preg_replace("/\\\\.{1}/", '', $val);
661:         $local = preg_replace("/\"[^\"]+\"/", '', $local);
662: 
663:         /* validate remaining unescaped characters */
664:         if (preg_match("/[[:print:]]/", $local) && !preg_match("/[@\\\",\[\]]/", $local)) {
665:             return true;
666:         }
667:         return false;
668:     }

671:         /* check for a dot, max allowed length and standard ASCII characters */
672:         if (strpos($val, '.') === false || strlen($val) > 255 || preg_match("/[^A-Z0-9\-\.]/i", $val) ||
673:             $val{0} == '-' || $val{(strlen($val) - 1)} == '-') {
674:             return false;

679:     function encode_header_fld($input, $email=true) {
680:         $res = array();
681:         $input = trim($input, ',; ');
682:         if (strstr($input, ' ')) {
683:             $parts = explode(' ', $input);
684:         }
685:         else {
686:             $parts[] = $input;
687:         }
688:         foreach ($parts as $v) {
689:             if (preg_match('/(?:[^\x00-\x7F])/',$v) === 1) {
690:                 $leading_quote = false;
691:                 $trailing_quote = false;
692:                 if (substr($v, 0, 1) == '"') {
693:                     $v = substr($v, 1);
694:                     $leading_quote = true;
695:                 }
696:                 if (substr($v, -1) == '"') {
697:                     $trailing_quote = true;
698:                     $v = substr($v, 0, -1);
699:                 }
700:                 $enc_val = '=?UTF-8?B?'.base64_encode($v).'?=';
701:                 if ($leading_quote) {
702:                     $enc_val = '"'.$enc_val;
703:                 }
704:                 if ($trailing_quote) {
705:                     $enc_val = $enc_val.'"';
706:                 }
707:                 $res[] = $enc_val;
708:             }
709:             else {
710:                 if ($email && strpos($v, '@') !== false && $this->match_email_full($v)) {
711:                     $res[] = '<'.$v.'>';
712:                 }
713:                 else {
714:                     $res[] = $v;
715:                 }
716:             }
717:         }
718:         $string = preg_replace("/\s{2,}/", ' ', trim(implode(' ', $res)));
719:         return $string;
720:     }

731:             $v = str_replace(array(',', ';'), array(' , ', ' ; '), $v);
